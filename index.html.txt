<script>

/* Start Music On First Click */
document.body.addEventListener("click",function(){
document.getElementById("bgMusic").play();
},{once:true});

/* Cursor Hearts */
document.addEventListener("mousemove",function(e){
let heart=document.createElement("div");
heart.className="cursor-heart";
heart.innerHTML="💖";
heart.style.left=e.pageX+"px";
heart.style.top=e.pageY+"px";
document.body.appendChild(heart);
setTimeout(()=>heart.remove(),1000);
});

/* Page Control */
function showPage(id){
document.querySelectorAll(".page").forEach(p=>p.classList.remove("active"));
document.getElementById(id).classList.add("active");
}

function nextPage(id){
showPage(id);
startFlowers();
}

/* Answer Check */
function checkAnswer(){
let ans=document.getElementById("answer").value.toLowerCase().trim();
if(ans==="book"){
showPage("card");
startFlowers();
}else{
showPage("wrong");
}
}

/* Love Yes */
function loveYes(){

}

/* Love No Button Runs */
let noBtn=document.getElementById("runNo");
noBtn.addEventListener("mouseover",function(){
noBtn.style.position="absolute";
noBtn.style.left=Math.random()*80+"vw";
noBtn.style.top=Math.random()*80+"vh";
});

/* Flowers */
function startFlowers(){
for(let i=0;i<40;i++){
let f=document.createElement("div");
f.className="flower";
f.innerHTML=Math.random()>0.5?"🌸":"🌺";
f.style.left=Math.random()*100+"vw";
f.style.animationDuration=(Math.random()*4+3)+"s";
document.body.appendChild(f);
setTimeout(()=>f.remove(),7000);
}
}

/* Fireworks */
function startFireworks(){
for(let i=0;i<30;i++){
let fw=document.createElement("div");
fw.className="firework";
fw.style.top=Math.random()*100+"%";
fw.style.left=Math.random()*100+"%";
document.getElementById("final").appendChild(fw);
}
}
